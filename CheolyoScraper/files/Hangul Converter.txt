	function changeLocation(model) {
		var url = document.URL;
		
		var otherUrl = "";
		var currentType = "";
		
		var urlVarBookId = getUrlVars()["bookId"];
		if (urlVarBookId) {
			urlVarBookId = decodeURI(urlVarBookId);
			if (urlVarBookId != selectBookId) {
				url = decodeURI(url);
				url = url.replace(urlVarBookId, selectBookId);
			}
		}
		
		if (url.indexOf('types=o') &gt; -1 || url.indexOf('/o/') &gt; -1) {
			if (url.indexOf('types=o') &gt; -1) {
				otherUrl = url.replace('types=o', 'types=r');
				if (url.indexOf('/o/') &gt; -1) {
					otherUrl = otherUrl.replace('/o/', '/r/');
				}
			} else if (url.indexOf('/o/') &gt; -1) {
				otherUrl = url.replace('/o/', '/r/');
			}
			currentType = "o";
		} else if (url.indexOf('types=r') &gt; -1 || url.indexOf('/r/') &gt; -1) {
			if (url.indexOf('types=r') &gt; -1) {
				otherUrl = url.replace('types=r', 'types=o');
				if (url.indexOf('/r/') &gt; -1) {
					otherUrl = otherUrl.replace('/r/', '/o/');
				}
			} else if (url.indexOf('/r/') &gt; -1) {
				otherUrl = url.replace('/r/', '/o/');
			}
			currentType = "r";
		} else {
			otherUrl = url;
		}
		
		// 수정 : levelId 값 변경.
		var paramItemId = getUrlVars()["itemId"];	// 아이템ID

    function setToggleHandlerByType(type,id) {
    	var selector;
        
        if ( type == "series" ) {
            selector = "#view_series_" + id.replace("$", "\\$");
        } else {
            selector = "#" + id.replace("$","\\$");
        }
       
        if ( $(selector).length &gt; 0 &amp;&amp; $(selector).data("eventReady") != "ok" ) {
        	$(selector).data("eventReady", "ok");
        	if ( type == "series" ) {
                $(selector).find("&gt;dt").on("click", function(e) {
                    $(this).siblings("dd").toggle();
                });
            } else if ( type == "year" ) {
                $(selector).find("&gt;a").on("click", function(e) {
                    $(this).siblings("ul").toggle();
                });
            }
        }
    }

      				var queryString = url.substring(url.indexOf("?")+1);
      				if(queryString.indexOf("/") &gt; -1){
          				if(queryString.indexOf("kingYear") &gt; -1){
          					var kingYearQueryString = queryString.substring(queryString.indexOf("#"));
          					var kingYears = kingYearQueryString.split("/");

          					dateOccured = kingYears[4];
          					if(kingYears.length &gt; 5){
          						dateOccured +="-"+kingYears[5];

          							if($monLi){
          								dateOccured += "-" + $monLi.attr("id").replace("kingdate_", "");
          							}
          						}
          					}
          					$("#searchCommand").find("input[name='kingYear']").val(kingYear);
          					$("#searchCommand").find("input[name='dateOccured']").val(dateOccured);
          				}else if(queryString.indexOf("book-show") &gt; -1){

      					paramBookId = getUrlVars()["bookId"];
      					if(paramBookId){
      						$("#searchCommand").find("input[name='bookId']").val(paramBookId);
      					}
      				}
      			}
      		}

        	$("#searchCommand").attr('action', 'downloadItemLevel.do').submit();
